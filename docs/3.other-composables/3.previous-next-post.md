const route = useRoute()

</script>

    <UButton

      :alt="prevButton"
      :to="'/' + prev"
      icon="i-heroicons-arrow-left-16-solid"
      variant="soft"
      size="sm"
    />
  </div>
  <div class="flex justify-end">
    <UButton
      v-if="next"
      :alt="nextButton"
      :to="'/' + next"
      trailing-icon="i-heroicons-arrow-right-16-solid"
      variant="soft"
      size="sm"
    />
  </div>
</template>
